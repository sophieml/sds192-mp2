most_common_occupations <- individuals %>% 
  filter(!is.na(occupation)) %>%
  group_by(occupation) %>% 
  summarize(num_occ = n()) %>% 
  filter(num_occ > 10000) %>% 
  mutate(occ_type = occupation) %>%
